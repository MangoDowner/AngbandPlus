ADD_LIBRARY (${DBWIN_LIB}
    STATIC
    lib_dbwin.c
)
SET_TARGET_PROPERTIES (${DBWIN_LIB} PROPERTIES PREFIX "")
SET_PROPERTY (
	TARGET ${DBWIN_LIB}
	PROPERTY COMPILE_DEFINITIONS
	${PLATFORM}
)



SET (DBWIN_SHLIB dbwin)
ADD_LIBRARY (${DBWIN_SHLIB}
    SHARED
    tcl_dbwin.c
)
SET_TARGET_PROPERTIES (${DBWIN_SHLIB}
	PROPERTIES
	PREFIX ""
)
SET_PROPERTY (
	TARGET ${DBWIN_SHLIB}
	PROPERTY COMPILE_DEFINITIONS
	USE_TCL_STUBS
)
TARGET_LINK_LIBRARIES (${DBWIN_SHLIB}
    ${TCL_STUB_LIBRARY}
    ${DBWIN_LIB}
)

INSTALL (
	TARGETS ${DBWIN_SHLIB}
	RUNTIME DESTINATION lib/dbwin
)
