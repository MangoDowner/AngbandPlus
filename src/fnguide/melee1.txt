*** melee1.c ***

static int monster_critical(int dice, int sides, int dam)
  calculates a monster's chance of a critical hit based on the
  <dice> and <sides> of its attack and the <dam>age that this
  particular hit does
static int check_hit(int power, int level)
  determines if a monster hits the player based on its <power>
  and its <level>.
bool carried_make_attack_normal(int r_idx)
  the player's carried monster of type <r_idx> attacks in a
  certain way, returning success
bool make_attack_normal(int m_idx, byte divis)
  does a normal monster attack carried out by a monster of 
  type <m_idx>, with some <divis>ion applied, returning whether
  or not the hit happened
