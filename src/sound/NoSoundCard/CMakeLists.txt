SET_PROPERTY (
    DIRECTORY
    PROPERTY COMPILE_DEFINITIONS
	BUILD_icon_dll
	BUILD_sound_dll
	${PLATFORM}
)

INCLUDE_DIRECTORIES (
    ${COMMONTK}/src/common-dll
    ${DBWIN_HDRS}
)

SET (SOUND_DLL sound-nocard)

ADD_LIBRARY (${SOUND_DLL}
    SHARED
    nosoundcard.c
)

SET_TARGET_PROPERTIES (${SOUND_DLL}
    PROPERTIES PREFIX ""
)

TARGET_LINK_LIBRARIES (${SOUND_DLL}
    ${DBWIN_LIB}
    gdi32
)

INSTALL (
	TARGETS ${SOUND_DLL}
	RUNTIME DESTINATION lib
)
